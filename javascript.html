class Product { price; manufacturer;
constructor(price, manufacturer) { this.price = price; this.manufacturer = manufacturer; }
get price() { return this.price; }
set price() { if (price < 0.00) { console.log("Price cannot be negative."); } else { this.price = price; } }
get manufacturer() { return this.manufacturer; }
set manufacturer(manufacturer) { this.manufacturer = manufacturer; }
}
